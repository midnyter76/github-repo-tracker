PORTED MATH (translate this exact JS to Python; `now` is the run datetime):
  velFmt(v)      -> f"{v:.1f}"
  starsFull(n)   -> f"{n:,}"                         # en-US thousands separators
  ageStr(created_at, now):
      days = (now.date() - datetime.fromisoformat(created_at).date()).days
      -> "today" if days <= 0 else "1d old" if days == 1 else f"{days}d old"
  bucket_max_vel = max(1.0, *[e.velocity_per_day for e in entries])   # per-bucket, floor 1.0
  barPct(v, bucket_max_vel):
      pct = max(7, _jsround((v / bucket_max_vel) * 100))              # floor 7
  barFill(pct):
      hue = _jsround(152 - (pct / 100) * 80)                          # green(152)->amber(72)
      -> f"linear-gradient(90deg, oklch(0.6 0.12 {hue}), oklch(0.82 0.17 {hue}))"
  # JS Math.round is half-up; Python round() is banker's. Use a helper:
  #   def _jsround(x): return math.floor(x + 0.5)

  topMover = the entry with the MAX velocity_per_day across ALL buckets' entries. Carry the bucket
      title THROUGH the flatten so NO fragile back-search is needed. (The same repo can appear as
      DISTINCT dicts in multiple buckets with DIFFERENT velocities — rank._build_entry builds a new
      dict per bucket — so an identity/`==` back-search to recover the bucket is a latent bug.)
      tagged = [(title, e) for key, kicker, title in _HTML_SECTIONS for e in buckets[key]["entries"]]
      topMoverBucketTitle, topMover = (
          max(tagged, key=lambda t: t[1]["velocity_per_day"]) if tagged else (None, None)
      )   # max() returns the FIRST element on ties, matching JS `>` semantics.
      # If topMover is None (all buckets empty): omit the hero <a> and render a placeholder,
      # e.g. an italic line "No qualifying repos yet — the radar is still warming up." Never crash.
